%
% Tabular cell and link arrows
%

\newcommand\cell[2]{%
    \tikz[remember picture] \node[inner sep=0, outer sep=0] (#1){#2};%
}

\newcommand\around[1]{%
\begin{tikzpicture}[remember picture, overlay, shift={(0,0)}]
	\node [draw,circle,thick,densely dashed,fit=(#1),inner sep=1pt] {};
\end{tikzpicture}%
}

\newcommand\eround[2]{%
\begin{tikzpicture}[remember picture, overlay, shift={(0,0)}]
	\node [draw,rectangle,rounded corners,thick,densely dashed,fit=(#1) (#2),inner sep=3pt] {};
\end{tikzpicture}%
}

\newcommand\link[2]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed,->] (#1.east) edge (#2.west);
\end{tikzpicture}%
}

\newcommand\linkleft[3]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed,bend left=#3,->] (#1.east) edge (#2.west);
\end{tikzpicture}%
}

\newcommand\linkright[3]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed,bend right=#3,->] (#1.east) edge (#2.east);
\end{tikzpicture}%
}

\newcommand\linkdown[3]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
    \path [thick,dashed,bend left=10 ] (#1.east) edge (#2.west);
    \path [thick,dashed,bend left=10,->] (#2.west) edge (#3.east);
\end{tikzpicture}%
}

\newcommand\linkup[3]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
    \path [thick,dashed,bend right=10 ] (#1.east) edge (#2.west);
    \path [thick,dashed,bend right=10,->] (#2.west) edge (#3.east);
\end{tikzpicture}%
}

\newcommand\edge[2]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed] (#1.east) edge (#2.west);
\end{tikzpicture}%
}

\newcommand\edgedown[2]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed, bend right=40] (#1.east) edge (#2.west);
\end{tikzpicture}%
}

\newcommand\edgeup[2]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed, bend right=40] (#1.east) edge (#2.east);
\end{tikzpicture}%
}

\newcommand\strike[2]{%
\begin{tikzpicture}[remember picture, overlay, >=stealth, shift={(0,0)}]
	\path [thick,dashed] (#1.west) edge (#2.east);
\end{tikzpicture}%
}
